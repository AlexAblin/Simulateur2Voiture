1   package AppliSimu;
2   
3   import DomaineVoiture.Voiture;
4   
5   import javax.swing.*;
6   import java.awt.*;
7   import java.awt.event.ActionEvent;
8   import java.awt.event.ActionListener;
9   
10  public class CommandeVoiture extends JPanel implements ActionListener {
11  
12      private JButton boutonAccelerer;
13      private JButton boutonInverserDirection;
14      private JButton boutonTournerDroite;
15  	private JButton boutonTournerGauche;
16      private JButton boutonArreter;
17  	private Voiture maVoiture;
18  
19  	
20  	public CommandeVoiture(final JFrame fenetre,final Voiture maVoiture) {
21  		
22  		super();
23  		this.setLayout(new FlowLayout());
24   
25  		boutonAccelerer = new JButton("Accelerer");
26  		boutonAccelerer.addActionListener(this);
27  		this.add(boutonAccelerer);
28  
29  		boutonInverserDirection = new JButton("Changer direction");
30  		boutonInverserDirection.addActionListener(this);
31  		this.add(boutonInverserDirection);
32  
33          boutonTournerDroite = new JButton("Tourner à droite");
34          boutonTournerDroite.addActionListener(this);
35          this.add(boutonTournerDroite);
36  
37  		boutonTournerGauche = new JButton("Tourner à gauche");
38  		boutonTournerGauche.addActionListener(this);
39  		this.add(boutonTournerGauche);
40  
41  		boutonArreter = new JButton("Arreter");
42  		boutonArreter.addActionListener(this);
43  		this.add(boutonArreter);
44  
45          fenetre.add(this);
46  		this.maVoiture = maVoiture;
47         }
48  
49  
50          @Override
51  	public void actionPerformed(ActionEvent event) {
52          Object bouton = event.getSource();
53          if (bouton == boutonAccelerer){
54              maVoiture.accelerer();
55          }   else if (bouton == boutonInverserDirection) {
56              maVoiture.inverserDirection();
57          }   else if (bouton == boutonTournerDroite) {
58              maVoiture.tournerADroite();
59  		}   else if (bouton == boutonTournerGauche) {
60  			maVoiture.tournerAGauche();
61          }   else if (bouton == boutonArreter) {
62              maVoiture.arreter();
63          }
64  
65      }
66  }
